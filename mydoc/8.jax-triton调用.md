def register_custom_call_handler(

  xla_platform_name = xla_platform_names.get(platform, platform)
  with _custom_callback_lock:
    _custom_callback_handler[xla_platform_name] = handler

m.def(
      "register_custom_call_target",
      [](nb::capsule c_api, nb::object fn_name_py, nb::object fn,
         nb::str xla_platform_name, int api_version,
         XLA_FFI_Handler_Traits traits) {
        const char* fn_name_c_str;

    // The opaque data is a zlib compressed protobuf.
    JAX_ASSIGN_OR_RETURN(std::string serialized, ZlibUncompress(opaque));

    jax_triton::TritonAnyKernelCall proto;
